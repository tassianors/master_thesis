% lista de símbolos em ordem alfabética (opcional)
\begin{listofsymbols}{$H(q, \theta)$}
	\item [$G_0(z)$] Função de transferência que representa a planta real do sistema.
	\item [$G(q, \theta)$] Função de transferência que representa a planta a ser estimada na identificação.
	\item [$H_0(z)$] Função de transferência desconhecida que altera o ruído branco que atua sobre o sistema.
	\item [$H(q, \theta)$] Função de transferência que altera o ruído branco que atua sobre o sistema, a qual se quer
	identificar.
	\item [$\theta^*$] Representa o conjunto de parâmetros que faz com que o modelo identificado seja igual ao sistema
	real.
	\item [$\theta$] representa um conjunto de parâmetros estimado do sistema.
	\item [$\hat{\theta}_N$] representa a estimativa para um certo valor de N pontos.
	\item [$\mathcal{S}$] Representa o sistema real sob análise.
	\item [$\mathcal{M}$] representa a classe de modelos utilizada para identificar o sistema real.
	\item [$T(z)$] representa o comportamento do sistema em malha fechada.
	\item [$T_d(z)$] representa o comportamento do sistema em malha fechada desejado.
	\item [$u(t)$] Sinal de saída do controlador ou entrada da planta.
	\item [$y(t)$] sinal de saída da planta.
	\item [$e(t)$] Ruido branco.
	\item [$\nu(t)$] ruído resultante depois que o ruídio branco é alterado por $H_0(z)$
	\item [$\phi$] variável de regressão utilizada para identificação do sistema.
	\item [$Z(t)$] instrumento utilizado no método de variáveis instrumentais.
	\item [$E(\cdot)$] representa a esperança de valor.
	\item [$\Phi$ ] representa o espectro de um sinal.
	\item [$\prod$] Produtório
	\item [$\sum$] Somatório
	\item [$L(z)$] Filtro utilizado no método VRFT.
	\item [$\sigma_e^2$] Variância do ruído.
	\item [$\chi^2$] Nivel de confiança 
\end{listofsymbols}
